let currentQuestion = 0;

const typeScores = { Decoder: 0, Designer: 0, Deliverer: 0, Resonator: 0 };


function showQuestion() {

  const q = quizData[currentQuestion];

  const quizDiv = document.getElementById('quiz');

  quizDiv.innerHTML = `<div class='question'><h2>${q.question}</h2></div>`;

  const answersDiv = document.createElement('div');

  answersDiv.className = 'answers';

  q.answers.forEach(a => {

    const button = document.createElement('button');

    button.innerText = a.text;

    button.onclick = () => selectAnswer(a.type);

    answersDiv.appendChild(button);

  });

  quizDiv.appendChild(answersDiv);

}


function selectAnswer(type) {

  typeScores[type]++;

  currentQuestion++;

  if (currentQuestion < quizData.length) {

    showQuestion();

  } else {

    showResult();

  }

}


function showResult() {

  document.getElementById('quiz').style.display = 'none';

  const resultDiv = document.getElementById('result');

  const topType = Object.keys(typeScores).reduce((a, b) => typeScores[a] > typeScores[b] ? a : b);

  resultDiv.innerHTML = `<h2>You are a ${topType}.</h2>

    <p>Enter your email to receive your full Signal Profile and mythic tools.</p>

    <input type='email' placeholder='Your email' id='email'><br>

    <button class='submit' onclick='submitEmail()'>Activate</button>`;

  resultDiv.style.display = 'block';

}

}
